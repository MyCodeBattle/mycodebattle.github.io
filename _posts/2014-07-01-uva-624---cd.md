## 题意

输出让磁带装最多的歌曲的分钟数

## 思路

0-1背包，增加了输出解。因为我是顺序推过去的，就是逆序输出来。

    #include <cstdio>#include <cstring>#include <algorithm>using namespace std;const int MAXN = 15000;int d[MAXN], CD[25], vis[25][MAXN];int main(){    //freopen("input.txt", "r", stdin);    int T, i, j, n, total;    while (~scanf("%d", &total))    {        scanf("%d", &n);        for (i = 1; i <= n; i++)            scanf("%d", &CD[i]);        memset(d, 0, sizeof(d));        memset(vis, 0, sizeof(vis));        for (i = 1; i <= n; i++)        {            for (j = total; j >= CD[i]; j--)            {                if (d[j] < d[j - CD[i]] + CD[i])                {                    d[j] = d[j - CD[i]] + CD[i];                    vis[i][j] = 1;                }            }        }        for (i = n, j = d[total]; i >= 1 && j > 0; i--)        {            if (vis[i][j])            {                printf("%d ", CD[i]);                j -= CD[i];            }        }        printf("sum:%d\n", d[total]);    }    return 0;}
