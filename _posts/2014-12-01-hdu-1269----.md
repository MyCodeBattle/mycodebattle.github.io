int SCCCnt, DFSClock, pre[MAXN], lowlink[MAXN], sccno[MAXN];
stack<int> st;
vector<int> G[MAXN];
 
void DFS(int from)
{
    pre[from] = lowlink[from] = ++DFSClock;
    st.push(from);
    for (int i = 0; i < SZ(G[from]); i++)
    {
        int to = G[from][i];
        if (!pre[to])
        {
            DFS(to);
            lowlink[from] = min(lowlink[from], lowlink[to]);
        }
        else if (!sccno[to]) lowlink[from] = min(lowlink[from], pre[to]);
    }
    if (lowlink[from] == pre[from])
    {
        SCCCnt++;
        while (1)
        {
            int x = st.top(); st.pop();
            sccno[x] = SCCCnt;
            if (x == from) break;
        }
    }
}
 
void FindSCC(int n)
{
    DFSClock = SCCCnt = 0;
    MS(sccno, 0); MS(pre, 0);
    for (int i = 1; i <= n; i++)
        if (!pre[i]) DFS(i);
}
 
int main()
{
    //ROP;
    int n, m, i, j;

    {
        for (int i = 0; i <= n; i++) G[i].clear();
        for (i = 0; i < m; i++)
        {
            int from, to;
            scanf("%d%d", &from, &to);
            G[from].PB(to);
        }
        FindSCC(n);
        int tmp = sccno[1];
        for (i = 2; i <= n; i++)
            if (sccno[i] != tmp) break;
        printf("%s\n", i == n + 1 ? "Yes" : "No");
    }
    return 0;
}
